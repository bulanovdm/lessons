var  - 1) область видимости функция или весь вскрипт если переменная глобальная
       2) объявления переменных var производится в начале функции (или скрипта для глобальных переменных).


 1) пример:
   if (true) {
      var test = true; // используем var вместо let
   }
   alert(test); // true т.к. блок if не ограничивает область видимости ( так же цикл не ограничивает облать видимости

 2) объявление var происходит всегда вначале скрипта, где бы var не находился :
   function sayHi() {
     phrase = "Привет";

     if (false) {
       var phrase;  // объявление var происходит в начале скрипта
     }

     alert(phrase);
   }
   sayHi();  // "Привет"

   объявление var произойдёт в начале скрипта, но присвоение останется на своём месте
   function sayHi() {
     alert(phrase);

     var phrase = "Привет";
   }
   sayHi(); // undefined





let  - область видимости ограничена скобками
  пример:
  if (true) {
     let test = true; // используем let
  }
